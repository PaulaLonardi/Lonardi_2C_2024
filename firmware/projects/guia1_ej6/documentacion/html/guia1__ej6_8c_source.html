107 {
108 uint8_t mask = 1; //mascara que se usa en la comparacion

117 //recorro el valor para ver cual está en 0 o 1 y pongo la pata en bajo o alto respectivamente

120 if ((mask & valor) != 0){

123 GPIOOn(p_vertor_pines[j].pin);
