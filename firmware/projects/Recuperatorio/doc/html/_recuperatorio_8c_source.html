96void tarea_medir(void *pvParameter)
97{
98 uint16_t distancia_anterior = 10;
99 velocidad_maxima = 0;
100 while (1)
101 {
102 distancia_anterior = distancia;
103 distancia = HcSr04ReadDistanceInCentimeters() / 100; // lo paso a metros
104 //distancia_anterior = distancia;
105 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /* La tarea espera en este punto hasta recibir una notificación */
106
107 // pesaje = false;
108
109 if (distancia < 10)
110 {
111 // calculo la velocidad
112 velocidad = (distancia_anterior - distancia) / 0.1;
113
114 if (velocidad > 8) // velocidad mayor a 8m/s: LED3,
115 {
116 LedOff(LED_1);
117 LedOff(LED_2);
118 LedOn(LED_3);
119 pesaje = false;
120 if (velocidad > velocidad_maxima)
121 {
122 velocidad_maxima = velocidad;
123 }
124 }
125 else if (velocidad < 8 && velocidad > 0) // velocidad entre 0m/s y 8m/s: LED2,
126 {
127 LedOff(LED_1);
128 LedOn(LED_2);
129 LedOff(LED_3);
130 pesaje = false;
131
132 if (velocidad > velocidad_maxima)
133 {
134 velocidad_maxima = velocidad;
135 }
136 }
137 if (velocidad == 0) // vehículo detenido:LED1.
138 {
139 LedOn(LED_1);
140 LedOff(LED_2);
141 LedOff(LED_3);
142
143 pesaje = true;
144 }
145 }
146 }
147}

159void tarea_pesar(void *pvParameter)
160{
161 uint16_t peso_anterior_g1 = 0;
162 uint16_t peso_anterior_g2 = 0;
163
164 uint16_t iteracion;
165 iteracion = 1;
166 while (1)
167 {
168 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /* La tarea espera en este punto hasta recibir una notificación */
169
170 if (pesaje)
171 {
172 peso_anterior_g1 = peso_g1;
173 peso_anterior_g2 = peso_g2;
174 AnalogInputReadSingle(CH1, &peso_g1);
175 AnalogInputReadSingle(CH2, &peso_g2);
176 iteracion++;
177
178 if (iteracion <= 50)
179 {
180 peso_g1 = peso_g1 + peso_anterior_g1;
181 peso_g2 = peso_g2 + peso_anterior_g2;
182 }
183 else
184 {
185 promedio_peso = ((peso_g1 + peso_g2) / 50)*(20000/3.3); //realizo la conversion a kilos
186 iteracion = 0;
187 UartSendString(UART_PC, "Peso: ");
188 UartSendString(UART_PC, (const char *)UartItoa(promedio_peso, 10));
189 UartSendString(UART_PC, " kg\r\n");
190
191 UartSendString(UART_PC, "Velocidad maxima: ");
192 UartSendString(UART_PC, (const char *)UartItoa(velocidad_maxima, 10));
193 UartSendString(UART_PC, " m/s\r\n");
194 }
195 }
196 }
